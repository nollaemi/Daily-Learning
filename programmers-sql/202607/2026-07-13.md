BOOK 테이블에서 2021년에 출판된 '인문' 카테고리에 속하는 도서 리스트를 찾아서 도서 ID(BOOK_ID), 출판일 (PUBLISHED_DATE)을 출력하는 SQL문을 작성해주세요.

SELECT book_id, published_date
FROM BOOK
WHERE category = '인문' AND YEAR(published_date) = 2021
ORDER BY published_date;

SELECT book_id, published_date
FROM BOOK
WHERE (category = '인문') AND (published_date BETWEEN '2021-01-01' AND '2021-12-31')
ORDER BY published_date

SELECT book_id, published_date
FROM BOOK
WHERE category = '인문'
  AND CAST(published_date AS CHAR) LIKE '2021%'
ORDER BY published_date;

SELECT a.FLAVOR
FROM FIRST_HALF AS a, ICECREAM_INFO AS b
WHERE (a.TOTAL_ORDER > 3000) AND (b.INGREDIENT_TYPE = 'fruit_based')
ORDER BY a.TOTAL_ORDER DESC # 실패한 코드

SELECT *

FROM FIRST_HALF F JOIN ICECREAM_INFO I ON F.FLAVOR = I.FLAVOR # 같은 결과 냄
